\element{analysis}{
\begin{question}{AnalysisQ01}
    You are presented with a mathematical expression which supposedly results in
        a value for one of the physical properties of an object.
    The expression is given as
    \begin{equation*}
        \pi h \left( r_1^2 + r_1 r_2 + r_2^2 \right)
    \end{equation*}
    where $h$, $r_1$, and $r_2$ all have dimensionality of length.
    Which physical property is this most likely to represent?
    \begin{choices}
      \correctchoice{volume}
        \wrongchoice{distance}
        \wrongchoice{area}
        \wrongchoice{angle}
    \end{choices}
\end{question}
}
