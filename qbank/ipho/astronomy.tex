\element{ipho}{
\begin{question}{astronomy-Q01}
    The earth orbits the sun once a year and the moon orbits the earth about once a month.
    From the earth you can observe the changing phases of the moon.
    If an observer stands on the moon and looks at the earth,
        what would be the period of the phases of the earth seen by that observer?
    \begin{choices}
        \wrongchoice{same period as the phases of the moon}
        \wrongchoice{a little longer than the period for the phases of the moon}
        \wrongchoice{a little shorter than the period for the phases of the moon}
        \wrongchoice{about one year}
    \end{choices}
\end{question}
}

\element{ipho}{
\begin{question}{astronomy-Q02}
    A parsec is a unit of length used by astronomers to measure distance.
    If a parsec is 3.26 light years,
        and a light year is the distance that light travels in a year,
        how many meters are there in a parsec?
    [Speed of light is \SI{3e8}{\meter\per\second}]
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{3e7}{\meter}}
        \wrongchoice{\SI{3e14}{\meter}}
        \wrongchoice{\SI{3e15}{\meter}}
        \wrongchoice{\SI{3e16}{\meter}}
    \end{choices}
    \end{multicols}
\end{question}
}
