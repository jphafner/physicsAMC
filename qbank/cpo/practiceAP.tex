\begin{question}{ch05-mc-Q34}
    A free-body diagram includes vectors representing the
        individual forces acting on an object.
    Which of these quantities should \emph{not} appear
        on a free-body diagram?
    \begin{choices}
        \wrongchoice{tension of a rope}
      \correctchoice{mass times acceleration}

\element{ap}{
\begin{question}{ch02-mc-Q40}
    An object rolls along level ground to the right at constant speed.
    Must there be any forces pushing this object to the right?
    \begin{choices}
        \wrongchoice{Yes: the \emph{only} force that act must be to the right.}
        \wrongchoice{Yes: but there could also be a friction force to the left.}
        \wrongchoice{No: no forces can act to the right.}
      \correctchoice{No: while there can be forces acting, no net force \emph{must} act.}
        \wrongchoice{The answer depends on the speed of the object.}
    \end{choices}
\end{question}
}
