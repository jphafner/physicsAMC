\element{cpo-short}{
\begin{question}{ch21-short-q01}
    List the speed of sound in each of the following media in order
        from fastest to slowest.
    \begin{enumerate*}
        \item Water at \SI{20}{\degreeCelsius}
        \item Air at \SI{20}{\degreeCelsius}
        \item Helium at \SI{20}{\degreeCelsius}
        \item Steel at \SI{20}{\degreeCelsius}
        \item Air at \SI{0}{\degreeCelsius}
    \end{enumerate*}
    \AMCOpen{lines=3}{
        \wrongchoice[W]{w}\scoring{0}
        \wrongchoice[P]{p}\scoring{1}
        \correctchoice[C]{c}\scoring{2}
    }
    %% ANS: D, A, C, B, E
\end{question}
}
