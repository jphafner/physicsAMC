\element{sat-mc}{
\begin{question}{math-t01-q02}
    For $i=\sqrt{-1}$, what is the sum of $\left(7+3i\right) + \left(-8+9i\right)$?
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{$-1 + 12i$}
        \wrongchoice{$-1 - 6i$}
        \wrongchoice{$15 + 12i$}
        \wrongchoice{$15 - 6i$}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{sat-mc}{
\begin{question}{math-t01-q03}
    On Saturday afternoon, Armand sent $m$ text messages each hour for 5 hours,
        and Tyrone sent $p$ text messages each hour for 4 hours.
    Which of the following represents the total number of messages sent by Armand and Tyrone on Saturday afternoon?
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{$9 mp$}
        \wrongchoice{$20 mp$}
        \wrongchoice{$5m + 4p$}
        \wrongchoice{$4m + 5p$}
    \end{choices}
    \end{multicols}
\end{question}
}
