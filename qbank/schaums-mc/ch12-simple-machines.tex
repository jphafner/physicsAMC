\element{schaums-mc}{
\begin{questionmult}{ch12-Q01}
    A machine can increase which one or more of the following?
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{work}
      \correctchoice{speed}
      \correctchoice{force}
      \correctchoice{torque}
    \end{choices}
    \end{multicols}
\end{questionmult}
}

\element{schaums-mc}{
\begin{question}{ch12-Q02}
    %% NOTE: remove acroyns
    A machine has an IMA of \num{6.0} and an AMA of \num{5.0}. 
    The work input needed by the machine to raise a \SI{40}{\kilo\gram} load by \SI{3.0}{\meter} is:
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{0.98}{\kilo\joule}}
        \wrongchoice{\SI{1.2}{\kilo\joule}}
      \correctchoice{\SI{1.4}{\kilo\joule}}
        \wrongchoice{\SI{5.9}{\kilo\joule}}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{schaums-mc}{
\begin{question}{ch12-Q03}
    A person uses a force of \SI{300}{\newton} to pry up one end of a \SI{120}{\kilo\gram} box with a lever \SI{2.4}{\meter} long. 
    The distance from the end of the box to the fulcrum is:
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{40}{\centi\meter}}
      \correctchoice{\SI{49}{\centi\meter}}
        \wrongchoice{\SI{81}{\centi\meter}}
        \wrongchoice{\SI{171}{\centi\meter}}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{schaums-mc}{
\begin{question}{ch12-Q04}
    The smallest number of pulleys needed for an IMA of 5 is:
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\num{3}}
      \correctchoice{\num{4}}
        \wrongchoice{\num{5}}
        \wrongchoice{\num{6}}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{schaums-mc}{
\begin{question}{ch12-Q05}
    A force of \SI{50}{\newton} is needed to raise a \SI{240}{\newton} load with a pulley system. 
    The load goes up \SI{1}{\meter} for every \SI{5}{\meter} of rope pulled through the pulleys. 
    The efficiency of the system is:
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{48}{\percent}}
        \wrongchoice{\SI{50}{\percent}}
      \correctchoice{\SI{96}{\percent}}
        \wrongchoice{\SI{104}{\percent}}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{schaums-mc}{
\begin{question}{ch12-Q06}
    Two sets of spur gears, each set 96 percent efficient,
        drive a worm gear that is 80 percent efficient. 
    The overall efficiency of the system is:
    \begin{multicols}{2}
    \begin{choices}
      \correctchoice{\SI{74}{\percent}}
        \wrongchoice{\SI{77}{\percent}}
        \wrongchoice{\SI{80}{\percent}}
        \wrongchoice{\SI{96}{\percent}}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{schaums-mc}{
\begin{question}{ch12-Q07}
    A screw jack has a pitch of \SI{4.0}{\milli\meter} and a handle \SI{50}{\centi\meter} long. 
    Its efficiency is 20 percent. 
    The force that must be applied to the handle in order to lift a load of \SI{400}{\kilo\gram} is:
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{1}{\newton}}
        \wrongchoice{\SI{2.5}{\newton}}
        \wrongchoice{\SI{5}{\newton}}
      \correctchoice{\SI{25}{\newton}}
    \end{choices}
    \end{multicols}
\end{question}
}
