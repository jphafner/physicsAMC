\element{schaums-mc}{
\begin{question}{ch24-Q01}
    The same volume of copper can have any of the following combinations of length $L$ and cross-sectional area $A$.
    Which combination has the most resistance?
    \begin{choices}
        \wrongchoice{$L$ and $A$}
        \wrongchoice{$L/2$ and $2A$}
      \correctchoice{$2L$ and $A/2$}
        \wrongchoice{All have the same resistance because the volume of copper is the same.}
    \end{choices}
\end{question}
}
