\element{schaums-mc}{
\begin{question}{ch26-Q01}
    The farad is not equivalent to which of the following combinations of units?
    \begin{choices}
        \wrongchoice{Coulomb squared per joule (\si{\coulomb\squared\per\joule})}
        \wrongchoice{Coulomb per volt (\si{\coulomb\per\volt})}
      \correctchoice{Coulomb volt squared (\si{\coulomb\volt\squared})}
        \wrongchoice{Joule per volt squared (\si{\joule\per\volt\squared})}
    \end{choices}
\end{question}
}
