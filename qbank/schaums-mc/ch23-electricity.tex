\element{schaums-mc}{
\begin{questionmult}{ch23-Q01}
    Which one or more of the following statements is true?
    \begin{choices}
      \correctchoice{All protons have the same charge.}
        \wrongchoice{Electrons and protons have equal masses.}
        \wrongchoice{Protons and neutrons have equal masses.}
      \correctchoice{Atomic nuclei contain only protons and neutrons.}
    \end{choices}
\end{questionmult}
}

\element{schaums-mc}{
\begin{question}{ch23-Q02}
    Electric charge occurs only in separate parcels:
    \begin{choices}
      \correctchoice{of \SI{\pm 1.6e-19}{\coulomb}}
        \wrongchoice{of \SI{\pm 1}{\coulomb}}
        \wrongchoice{whose value is proportional to the mass of the particle carrying the charge}
        \wrongchoice{that are different for positive and negative particles}
    \end{choices}
\end{question}
}
