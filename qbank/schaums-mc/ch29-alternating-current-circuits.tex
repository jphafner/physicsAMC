\element{schaums-mc}{
\begin{questionmult}{ch29-Q01}
    The current in an ac circuit:
    \begin{choices}
        \wrongchoice{leads the voltage}
        \wrongchoice{lags behind the voltage}
        \wrongchoice{is in phase with the voltage}
      \correctchoice{is any of the provided, depending on the circuit}
    \end{choices}
\end{questionmult}
}
