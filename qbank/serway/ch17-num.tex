\element{serway-num}{
\begin{questionmultx}{serway-ch17-q46}
    A bat, flying at \SI{5.00}{\meter\per\second}, emits a chirp at \SI{40.0}{\kilo\hertz}.
    If this sound pulse is reflected by a wall,
        what is the frequency of the echo received by the bat?
    ($v_{\text{sound}}=\SI{340}{\meter\per\second}$.)
    %% ANSWER:  \SI{41.2}{\kilo\hertz}
    \AMCnumericChoices{4.12}{
        vertical=false,
        digits=3,decimals=2,sign=true,Tsign=\hspace{1ex}\Large a,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
    \AMCnumericChoices{4}{
        vertical=false,
        digits=2,decimals=0,sign=true,Tsign=\hspace{1ex}\Large b,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
\end{questionmultx}
}
