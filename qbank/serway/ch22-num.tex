\element{serway-num}{
\begin{questionmultx}{serway-ch22-q49}
    In a nuclear power plant, the reactor produces \SI{500}{\degreeCelsius} steam that is used to power the steam turbines which generate \SI{1500}{\mega\watt} of electrical power.
    The cooling tower eliminates the waste heat at \SI{50}{\degreeCelsius}. 
    If the efficiency of the plant were that of a Carnot engine,
        at what rate would waste heat be vented to the atmosphere?
    %% ANSWER:  \SI{1080}{\mega\watt}
    \AMCnumericChoices{1.08}{
        vertical=false,
        digits=3,decimals=2,sign=true,Tsign=\hspace{1ex}\Large a,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
    \AMCnumericChoices{6}{
        vertical=false,
        digits=2,decimals=0,sign=true,Tsign=\hspace{1ex}\Large b,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
\end{questionmultx}
}

\element{serway-num}{
\begin{questionmultx}{serway-ch22-q50}
    Every second at Niagara Falls, some \SI{5000}{\meter\cubed} of water falls a distance of 50 m.
    What is the increase in entropy per second due to the falling water? 
    (Assume a \SI{20}{\degreeCelsius} environment).

        digits=2,decimals=0,sign=true,Tsign=\hspace{1ex}\Large b,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
\end{questionmultx}
}
