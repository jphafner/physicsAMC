\element{serway-num}{
\begin{questionmultx}{serway-ch05-q110}
    A high-diver of mass \SI{70}{\kilo\gram} jumps off a board 10 m above the water. 
    If, \SI{2}{\second} after entering the water his downward motion is stopped,
        what average upward force did the water exert on him?
    %% ANSWER: \SI{1180}{\newton}
    \AMCnumericChoices{1.18}{
        vertical=false,
        digits=3,decimals=2,sign=true,Tsign=\hspace{1ex}\Large a,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
    \AMCnumericChoices{3}{
        vertical=false,
        digits=2,decimals=0,sign=true,Tsign=\hspace{1ex}\Large b,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
\end{questionmultx}
}
