\element{serway-num}{
\begin{questionmultx}{serway-ch07-q77}
    While running, a person dissipates about \SI{0.6}{\joule} of mechanical energy per step per kilogram of body mass. 
    If a \SI{60}{\kilo\gram} person runs with a power of \SI{70}{\watt} during a race,
        how fast is the person running? 
    Assume a running step is \SI{1.5}{\meter} long.
    %% ANSWER:  \SI{2.92}{\meter\per\second}
    \AMCnumericChoices{2.92}{
        vertical=false,
        digits=3,decimals=2,sign=true,Tsign=\hspace{1ex}\Large a,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
    \AMCnumericChoices{0}{
        vertical=false,
        digits=2,decimals=0,sign=true,Tsign=\hspace{1ex}\Large b,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
\end{questionmultx}
}
