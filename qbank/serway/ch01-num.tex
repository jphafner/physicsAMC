\element{serway-num}{
\begin{question}{serway-ch01-q21}
    The standard kilogram is a platinum-iridium cylinder \SI{39}{\milli\meter} in height and \SI{39}{\milli\meter} in diameter. 
    What is the density of the material?
    %% ANSWER: \SI{21 475}{\kilo\gram\per\meter\cubed}
    \AMCOpen{lines=3}{
        \wrongchoice[W]{w}\scoring{0}
        \wrongchoice[P]{p}\scoring{1}

\end{question}
}
