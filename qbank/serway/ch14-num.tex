\element{serway-num}{
\begin{questionmultx}{serway-ch14-q47}
    One hundred milliliters of water is poured into a U-tube that has a cross-sectional area of \SI{1}{\centi\meter\squared}. 
    Then 100 milliliters of oil,
        with a density \SI{80}{\percent} that of water,
        is poured down one side of the U-tube so that the oil floats on the water.
    Find the difference in height of the liquid surfaces on the two sides of the U-tube.
    %% ANSWER:  \SI{20}{\centi\meter}
    \AMCnumericChoices{6.65}{
        vertical=false,
        digits=3,decimals=2,sign=true,Tsign=\hspace{1ex}\Large a,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
    \AMCnumericChoices{-11}{
        vertical=false,
        digits=2,decimals=0,sign=true,Tsign=\hspace{1ex}\Large b,
        borderwidth=0pt,backgroundcol=white,approx=5
    }
\end{questionmultx}
}
