\begin{questionmultx}{serway-ch18-q47}
    A student wants to establish a standing wave on a wire \SI{1.8}{\meter} long clamped at both ends.

    What is the minimum frequency she should apply to set up standing waves?

    \AMCnumericChoices{1.5}{

        vertical=false,

\end{questionmultx}
}
