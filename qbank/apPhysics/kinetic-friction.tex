\element{ap}{
\begin{question}{kinetic-friction-q09}
    An object of mass $m$ is sliding on a horizontal surface at constant velocity $v$.
    If the coefficient of kinetic friction between the object and the surface is $\mu$,
        which of the following best describes the force $F$ being applied to the object?
    \begin{choices}
      \correctchoice{of magnitude $\mu mg$, in the direction of $v$.}
        \wrongchoice{of magnitude $\mu mg$, in the direction opposite $v$.}
        \wrongchoice{of magnitude $\mu mgv$, normal to the surface.}
        \wrongchoice{of magnitude $\mu mgv$, but the direction cannot be determined.}
        \wrongchoice{Not enough information is provided to determine the magnitude or direction.}
    \end{choices}
\end{question}
}
