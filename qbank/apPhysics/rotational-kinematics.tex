      \correctchoice{\SI{2\pi/5}{\radian\per\second}}
        \wrongchoice{\SI{5\pi/2}{\radian\per\second}}

        \wrongchoice{\SI{5\pi}{\radian\per\second}}

    \end{multicols}
\end{question}
