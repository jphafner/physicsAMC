\element{holt-mc}{
\begin{question}{ch12-mc-Q05}
    The Doppler effect occurs in all but which of the following situations?
    \begin{choices}
        \wrongchoice{A source of sound moves toward a listener}
        \wrongchoice{A listener moves toward a source of sound}
        \wrongchoice{A listener and source of sound remain at rest with respect to each other}
        \wrongchoice{A listener and a source of sound move toward or away from each other}
    \end{choices}
\end{question}
}

\element{holt-mc}{

\element{holt-mc}{

    What can a dog hear a sound produced by a dog whistle,

    \begin{choices}

        \wrongchoice{Dogs detect sounds of higher frequency than do humans.}

        \wrongchoice{Dogs detect sounds of higher speed than do humans.}

\end{question}

\element{holt-mc}{

        \wrongchoice{\SI{2.0e14}{\meter}}
    \end{choices}
    \end{multicols}
\end{question}
}
