\element{holt-mc}{
\begin{question}{ch13-mc-Q01}
    Which equation is correct for calculating the focal point of a spherical mirror?
    Let $f$ represent the focal length,
        $p$ the object distance,
        and $q$ the image distance.
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{$\frac{1}{f} = \frac{1}{p} - \frac{1}{q}$}
        \wrongchoice{$\frac{1}{f} = \frac{1}{p} + \frac{1}{q}$}
        \wrongchoice{$\frac{1}{p} = \frac{1}{f} + \frac{1}{q}$}
        \wrongchoice{$\frac{1}{q} = \frac{1}{f} + \frac{1}{p}$}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{holt-mc}{

        speed of gamma rays and radio waves in a vacuum?

        \wrongchoice{Radio waves travel faster than gamma rays}
        \wrongchoice{Gamma rays and radio waves travel at the same speed in a vacuum}
        \wrongchoice{The speed of gamma rays and radio waves in a vacuum depends on their frequencies}
    \end{choices}
\end{question}
}

\element{holt-mc}{
\begin{question}{ch13-mc-Q03}
    Which of the following correctly states the law of reflection?
    \begin{choices}
        \wrongchoice{The angle between an incident ray of light and the normal to the mirror's surface equals the angle between the mirror's surface and the reflected light ray.}
        \wrongchoice{The angle between an incident ray of light and the normal to the mirror's surface equals the normal to the mirror's surface and the reflected light ray.}
        \wrongchoice{The angle between an incident ray of light and the normal to the mirror's surface equals the angle between the normal to the mirror's surface and the reflected light ray.}
        \wrongchoice{The angle between an incident ray of light and the normal to the mirror's surface is complementary to the angle between the normal and the reflected light ray.}
    \end{choices}
\end{question}
}
