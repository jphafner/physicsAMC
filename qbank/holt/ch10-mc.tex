\element{holt-mc}{
\begin{question}{holt-ch10-Q01}
    If there is no change in the internal energy of a gas,
        even though energy is transferred to the gas as heat and work,
        what is the thermodynamic process that the gas undergoes called?
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{adiabatic}
      \correctchoice{isothermal}
        \wrongchoice{isovolumetric}
        \wrongchoice{isobaric}
    \end{choices}
    \end{multicols}
\end{question}
}

\begin{question}{holt-ch10-Q02}

        which thermodynamics property does \emph{not} need to be know?
    \begin{choices}
        \wrongchoice{the energy transferred as heat to the engine}
        \wrongchoice{the energy transferred as heat from the engine}
      \correctchoice{the change in the interanal energy of the engine}

\end{question}
