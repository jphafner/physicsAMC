\element{holt-mc}{
\begin{question}{ch19-mc-Q01}
    Which of the following statements best describes the domains in unmagnetized iron?
    \begin{choices}
        \wrongchoice{There are no domains.}
        \wrongchoice{There are domains, but the domains are smaller than in magnetized iron.}
        \wrongchoice{There are domains, but the domains are oriented randomly.}
        \wrongchoice{There are domains, but the domains are not magnetized.}
    \end{choices}
\end{question}
}
