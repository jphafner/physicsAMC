\element{numeric}{
\begin{questionmultx}{free-fall-A-Q01}
\luaexec{
    %% Question
    local Q = [[
        A man is standing on top of a
            \string\SI{\%d}{\string\meter} tower.
        He throws a baseball straight up at a speed of
            \string\SI{\%0.1f}{\string\meter\string\per\string\second}.
        How long does it take to reach the ground
            divided by \string\SI{1}{\string\second}?
        [Ignore air resistance,
            $g=\string\SI{9.80665}{\string\meter\string\per\string\second\string\squared}$]
    ]]
    %% Random Values
    local h = math.random(50,85)
    local v = 10 + round(20*math.random(),1)
    local g = 9.80665
    local ans = (v + math.sqrt( (v*v) + (2*g*h) ) ) / g
    %% Print Question
    tex.print( string.format(Q, h, v) )
    %% Print AMCnumeric
    options = [[ digits=4, decimals=2, sign=True, approx=1 ]]
    tex.print( BeginCenter )
    tex.print( string.format(AMCnumeric, ans, options ) )
    tex.print( EndCenter )
}
\end{questionmultx}
}
