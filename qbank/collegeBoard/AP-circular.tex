\element{AP}{
\begin{question}{Jacobs-Q13}
    Which of the following must be true of an object in
        uniform circular motion?
    \begin{choices}
        \wrongchoice{Its velocity must be constant}
        \wrongchoice{Its acceleration and its velocity must
            be in opposite directions.}
      \correctchoice{Its acceleration and its velocity must
            be perpendicular to each other.}
        \wrongchoice{It must experience a force away from
            the center of the circle.}
        \wrongchoice{Its acceleration must be negative.}
    \end{choices}
\end{question}
}

\element{AP}{
\begin{question}{2004-APB-Q42}
    A child has a toy tied to the end of a string and whirls the toy
        at constant speed in a horizontal circular path of radius $R$.
    The toy completes each revolution of its motion in a time period $T$.
    What is the magnitude of the acceleration of the toy?
    \begin{choices}
        \wrongchoice{Zero}
      \correctchoice{$\dfrac{4 \pi^2 R}{T^2}$}
        \wrongchoice{$\dfrac{\pi R}{T^2}$}
        \wrongchoice{$g$}
        \wrongchoice{$2 \pi g$}
    \end{choices}
\end{question}
}
