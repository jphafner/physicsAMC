\element{AP}{
\begin{question}{Jacobs-Q08}
    A \SI{500}{\gram} block on a flat tabletop slides
        \SI{2.0}{\meter} to the right.
    If the coefficient of friction between the block
        and the table is \num{0.1}, how much
        work is done on the block by the table?
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{0.5}{\joule}}
      \correctchoice{\SI{1.0}{\joule}}
        \wrongchoice{zero}
        \wrongchoice{\SI{100}{\joule}}
        \wrongchoice{\SI{50}{\joule}}
    \end{choices}
    \end{multicols}
\end{question}
}
