\element{AP}{
\begin{question}{newton-Q01}
    A person standing on a horizontal floor feels two forces,
        the downward pull of gravity and the upward supporting force from the floor.
    These two forces:
    \begin{choices}
        \wrongchoice{have equal magnitudes and form an action/reaction pair.}
        \wrongchoice{have equal magnitudes but do not form an action/reaction pair.}
        \wrongchoice{have unequal magnitudes and form an action/reaction pair.}
        \wrongchoice{have unequal magnitudes and do not form an action/reaction pair.}
        \wrongchoice{None of the provided.}
    \end{choices}
\end{question}
}
