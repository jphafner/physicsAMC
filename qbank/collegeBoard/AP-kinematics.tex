\element{AP}{
\begin{question}{Jacobs-Q01}
    Which of the following must be true of an object
        that is slowing down?
    \begin{choices}
      \correctchoice{Its acceleration and its velocity must be in opposite directions.}
        \wrongchoice{Its acceleration must be negative}
        \wrongchoice{Its velocity must be smaller than its acceleration}
        \wrongchoice{It must experience more than one force.}
        \wrongchoice{Its velocity must be negative.}
    \end{choices}
\end{question}
}

\element{AP}{
\begin{question}{Jacobs-Q02}
    A baseball is thrown straight up.
    It reaches a peak height of \SI{15}{\meter},
        measured from the ground, in a time of
        \SI{1.7}{\second}.
    Treating ``up'' as the positive direction,
        what is the acceleration of the ball
        when it reaches its peak height?
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{0}{\meter\per\second\squared}}
        \wrongchoice{\SI[retain-explicit-plus]{+8.8}{\meter\per\second\squared}}
        \wrongchoice{\SI[retain-explicit-plus]{-8.8}{\meter\per\second\squared}}
        \wrongchoice{\SI[retain-explicit-plus]{+9.8}{\meter\per\second\squared}}
      \correctchoice{\SI[retain-explicit-plus]{-9.8}{\meter\per\second\squared}}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{AP}{
\begin{question}{sample-AP1-Q03}
    A person driving a car suddenly applies the brakes.
    The car takes \SI{4}{\second} to come to rest while traveling
        \SI{20}{\meter} at constant acceleration.
    Can the speed of the car immediately before the brakes were
        applied be determined without first determining the car's acceleration?
    \begin{choices}
      \correctchoice{Yes, by determining the average speed while braking and doubling it.}
        \wrongchoice{Yes, by diving the distance (\SI{20}{\meter}) by the time (\SI{4}{\second}).}
        \wrongchoice{No, because the acceleration is needed to use standard equations such as $\Delta x = v_0 t + \tfrac{1}{2} a t^2$.}
        \wrongchoice{No, because the fundamental relationship that defines velocity contains acceleration.}
    \end{choices}
\end{question}
}
