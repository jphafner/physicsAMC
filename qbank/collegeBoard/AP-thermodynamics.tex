\element{AP}{
\begin{question}{2004-APB-Q28}
    An ideal gas may be taken from one state to another state with
        a difference pressure, volume, and temperature along
        several different paths.
    Quantities that will always be the same for this process,
        regardless of which path is taken, include which of the following?
    \begin{itemize}
        \item[I] The change in internal energy of the gas
        \item[II] The heat exchanged between the gas and its surroundings
        \item[III] The work done by the gas
    \end{itemize}
    \begin{multicols}{2}
    \begin{choices}
      \correctchoice{I only}
        \wrongchoice{II only}
        \wrongchoice{III only}
        \wrongchoice{II and III only}
        \wrongchoice{I, II and III}
    \end{choices}
    \end{multicols}
\end{question}
}

        of area $A$ and thickness $d$ represented below.

        \includegraphics[keepaspectratio]{2004-APB-Q56}
