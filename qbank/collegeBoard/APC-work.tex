\element{AP}{
\begin{question}{work-Q01}
    A force $\mathbf{F}$ of strength \SI{20}{\newton} acts on an object of mass \SI{3}{\kilo\gram} as it moves a distance of \SI{4}{\meter}.
    If $\mathbf{F}$ is perpendicular to the \SI{4}{\meter} displacement,
        the work it does is equal to?
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{0}{\joule}}
        \wrongchoice{\SI{60}{\joule}}
        \wrongchoice{\SI{80}{\joule}}
        \wrongchoice{\SI{600}{\joule}}
        \wrongchoice{\SI{2400}{\joule}}
    \end{choices}
    \end{multicols}
\end{question}
}

\element{AP}{
\begin{question}{work-Q02}
    Under the influence of a force, an object of mass \SI{4}{\kilo\gram} accelerates from \SI{3}{\meter\per\second} to \SI{6}{\meter\per\second} in \SI{8}{\second}.
    How much work was done on the object during this time?
    \begin{multicols}{2}
    \begin{choices}
        \wrongchoice{\SI{27}{\joule}}
        \wrongchoice{\SI{54}{\joule}}
        \wrongchoice{\SI{72}{\joule}}
        \wrongchoice{\SI{96}{\joule}}
        \wrongchoice{Cannot be determined from the information given.}
    \end{choices}
    \end{multicols}
\end{question}
}
